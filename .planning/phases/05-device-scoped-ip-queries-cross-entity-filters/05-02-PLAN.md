# Plan 05-02: Cross-Entity Device Filters for IP Addresses and VLANs

## Objective

Add `device_name` filter to `list_ip_addresses` and `list_vlans` MCP tools so agents can scope IPAM queries by device — answering "what IPs/VLANs does device X use?" directly. Currently `nautobot_list_vlans` has no device filter at all, and `nautobot_list_ip_addresses` passes `device` to pynautobot but needs validation of the approach.

## must_haves

1. `nautobot_list_ip_addresses(device_name="X")` filters to IPs on device X's interfaces
2. `nautobot_list_vlans(device_name="X")` filters to VLANs on device X's interfaces
3. Both work via a correct Nautobot API approach (handling the M2M relationship properly)
4. Existing tests continue passing

<action>
Modify `list_ip_addresses` in `nautobot_mcp/ipam.py` to properly handle the `device` parameter. The current implementation passes `device` directly to `ip_addresses.filter()`, but Nautobot's IP address endpoint may not support direct device filtering.

**Strategy:** When `device` is provided, use a two-step approach:
1. First get all interface IDs for the device via `dcim.interfaces.filter(device=device_name)`
2. Then get IP-to-interface M2M records for those interfaces
3. Return the matched IPs

Update the `list_ip_addresses` function:

```python
def list_ip_addresses(
    client: NautobotClient,
    device: Optional[str] = None,
    interface: Optional[str] = None,
    prefix: Optional[str] = None,
    q: Optional[str] = None,
    limit: int = 0,
    **extra_filters: str,
) -> ListResponse[IPAddressSummary]:
    """List IP addresses with optional filtering."""
    try:
        # If device filter, use M2M approach to get IPs via interfaces
        if device:
            iface_records = list(client.api.dcim.interfaces.filter(device=device))
            seen_ip_ids = set()
            all_results = []
            for iface in iface_records:
                m2m_records = list(
                    client.api.ipam.ip_address_to_interface.filter(
                        interface=str(iface.id)
                    )
                )
                for m2m in m2m_records:
                    ip_id = str(m2m.ip_address.id)
                    if ip_id not in seen_ip_ids:
                        seen_ip_ids.add(ip_id)
                        ip_record = client.api.ipam.ip_addresses.get(id=ip_id)
                        if ip_record:
                            all_results.append(IPAddressSummary.from_nautobot(ip_record))
        else:
            filters = {}
            if interface:
                filters["interface"] = interface
            if prefix:
                filters["prefix"] = prefix
            if q:
                filters["q"] = q
            filters.update(extra_filters)

            if filters:
                records = list(client.api.ipam.ip_addresses.filter(**filters))
            else:
                records = list(client.api.ipam.ip_addresses.all())

            all_results = [IPAddressSummary.from_nautobot(r) for r in records]

        limited_results = all_results[:limit] if limit > 0 else all_results
        return ListResponse(count=len(all_results), results=limited_results)

    except Exception as e:
        client._handle_api_error(e, "list", "IPAddress")
        raise
```

Add `device` parameter to `list_vlans` in `nautobot_mcp/ipam.py`. When a device is specified, get the device's interfaces, extract VLAN IDs from tagged/untagged VLANs on those interfaces, then filter VLANs to that set.

```python
def list_vlans(
    client: NautobotClient,
    location: Optional[str] = None,
    tenant: Optional[str] = None,
    vlan_group: Optional[str] = None,
    vid: Optional[int] = None,
    device: Optional[str] = None,  # NEW parameter
    limit: int = 0,
    **extra_filters: str,
) -> ListResponse[VLANSummary]:
    """List VLANs with optional filtering."""
    try:
        if device:
            # Get interfaces for device, extract VLAN references
            iface_records = list(client.api.dcim.interfaces.filter(device=device))
            vlan_ids = set()
            for iface in iface_records:
                # Check untagged_vlan
                if hasattr(iface, "untagged_vlan") and iface.untagged_vlan:
                    vlan_ids.add(str(iface.untagged_vlan.id))
                # Check tagged_vlans
                if hasattr(iface, "tagged_vlans") and iface.tagged_vlans:
                    for vlan in iface.tagged_vlans:
                        vlan_ids.add(str(vlan.id) if hasattr(vlan, "id") else str(vlan))
            
            if not vlan_ids:
                return ListResponse(count=0, results=[])
            
            all_results = []
            for vlan_id in vlan_ids:
                record = client.api.ipam.vlans.get(id=vlan_id)
                if record:
                    all_results.append(VLANSummary.from_nautobot(record))
        else:
            filters = {}
            if location:
                filters["location"] = location
            if tenant:
                filters["tenant"] = tenant
            if vlan_group:
                filters["vlan_group"] = vlan_group
            if vid is not None:
                filters["vid"] = vid
            filters.update(extra_filters)

            if filters:
                records = list(client.api.ipam.vlans.filter(**filters))
            else:
                records = list(client.api.ipam.vlans.all())

            all_results = [VLANSummary.from_nautobot(r) for r in records]

        limited_results = all_results[:limit] if limit > 0 else all_results
        return ListResponse(count=len(all_results), results=limited_results)

    except Exception as e:
        client._handle_api_error(e, "list", "VLAN")
        raise
```
