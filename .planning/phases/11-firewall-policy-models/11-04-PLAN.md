def create_filter(device: str, name: str, family: str = "inet"):

@app.command("update-filter")
def update_filter(id: str, name: str = None, description: str = None):

@app.command("delete-filter")
def delete_filter(id: str):

# Full CRUD commands for FirewallPolicer:
@app.command("list-policers")
def list_policers(device: str, limit: int = 50):
    """List firewall policers for a device."""
    # Table columns: Name, Bandwidth, Burst, Logical, Actions (count)

@app.command("get-policer")
def get_policer(id: str):

@app.command("create-policer")
def create_policer(device: str, name: str, bandwidth_limit: int = None, bandwidth_unit: str = "bps"):

@app.command("update-policer")
def update_policer(id: str, ...):

@app.command("delete-policer")
def delete_policer(id: str):

# Read-only commands for sub-models:
@app.command("list-terms")
def list_terms(filter_id: str, limit: int = 50):

@app.command("get-term")
def get_term(id: str):
    """Get a firewall term with inlined match conditions and actions."""

@app.command("list-match-conditions")
@app.command("list-filter-actions")
@app.command("list-policer-actions")
