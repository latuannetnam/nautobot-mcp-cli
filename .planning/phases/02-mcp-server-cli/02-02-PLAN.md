1. Main Typer app: `app = typer.Typer(name="nautobot-mcp", help="Nautobot MCP CLI - Network automation via Nautobot API")`

2. Global callback with options:
```python
@app.callback()
def main(
    ctx: typer.Context,
    json_output: bool = typer.Option(False, "--json", help="Output as JSON"),
    profile: str = typer.Option(None, "--profile", help="Config profile name"),
    url: str = typer.Option(None, "--url", help="Nautobot URL override"),
    token: str = typer.Option(None, "--token", help="API token override"),
    no_verify: bool = typer.Option(False, "--no-verify", help="Skip SSL verification"),
):
    ctx.ensure_object(dict)
    ctx.obj["json"] = json_output
    ctx.obj["profile"] = profile
    ctx.obj["url"] = url
    ctx.obj["token"] = token
    ctx.obj["no_verify"] = no_verify
```

3. Helper to create client from context:
```python
def get_client_from_ctx(ctx: typer.Context) -> NautobotClient:
    from nautobot_mcp.config import NautobotSettings
    from nautobot_mcp.client import NautobotClient
    kwargs = {}
    if ctx.obj.get("url"): kwargs["url"] = ctx.obj["url"]
    if ctx.obj.get("token"): kwargs["token"] = ctx.obj["token"]
    settings = NautobotSettings(**kwargs)
    return NautobotClient(settings=settings)
```

4. Error handler that maps exceptions to exit codes:
```python
def handle_cli_error(e: Exception) -> None:
    import sys
    from nautobot_mcp.exceptions import (NautobotConnectionError, NautobotNotFoundError, NautobotValidationError)
    if isinstance(e, NautobotConnectionError):
        typer.echo(f"Connection error: {e.message}", err=True)
        raise typer.Exit(code=2)
    elif isinstance(e, NautobotNotFoundError):
        typer.echo(f"Not found: {e.message}", err=True)
        raise typer.Exit(code=3)
    elif isinstance(e, NautobotValidationError):
        typer.echo(f"Validation error: {e.message}", err=True)
        raise typer.Exit(code=4)
    else:
        typer.echo(f"Error: {str(e)}", err=True)
        raise typer.Exit(code=1)
```

Create 5 CLI command modules, all following the same pattern:

**nautobot_mcp/cli/devices.py:**
```python
import typer
from nautobot_mcp.cli.app import get_client_from_ctx, handle_cli_error
from nautobot_mcp.cli.formatters import output, DEVICE_COLUMNS
from nautobot_mcp import devices

devices_app = typer.Typer(help="Device operations")

@devices_app.command("list")
def devices_list(ctx: typer.Context, location: str = None, tenant: str = None, role: str = None, platform: str = None, q: str = None, limit: int = 50):
    try:
        client = get_client_from_ctx(ctx)
        result = devices.list_devices(client, location=location, tenant=tenant, role=role, platform=platform, q=q, limit=limit)
        output(result.model_dump(), ctx.obj.get("json", False), DEVICE_COLUMNS)
    except Exception as e:
        handle_cli_error(e)
```

Similarly: `get`, `create`, `update`, `delete` commands.

**nautobot_mcp/cli/interfaces.py:** `interfaces_app` with `list`, `get`, `create`, `update`, `assign-ip` commands.
