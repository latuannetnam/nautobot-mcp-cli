```python
from nautobot_mcp.parsers import ParserRegistry

@mcp.tool(
    name="nautobot_parse_config",
    description="Parse a device configuration JSON to extract structured network data. "
    "Supports JunOS (show configuration | display json). "
    "Returns interfaces, IPs, VLANs, routing instances, protocols, firewall filters, and system settings. "
    "Use network_os='juniper_junos' for Juniper devices."
)
async def nautobot_parse_config(
    config_json: str,
    network_os: str = "juniper_junos",
) -> dict:
    """Parse device config JSON into structured network data."""
    try:
        import json
        config_data = json.loads(config_json)
        parser = ParserRegistry.get(network_os)
        result = parser.parse(config_data)
        return result.model_dump()
    except ValueError as e:
        raise ToolError(str(e))
    except json.JSONDecodeError as e:
        raise ToolError(f"Invalid JSON: {str(e)}")
    except Exception as e:
        handle_error(e)
```

Also add a helper tool to list available parsers:
```python
@mcp.tool(
    name="nautobot_list_parsers",
    description="List available configuration parsers and their supported network_os identifiers."
)
async def nautobot_list_parsers() -> dict:
    """List registered config parsers."""
    return {"parsers": ParserRegistry.list_parsers()}
```

Create `nautobot_mcp/cli/commands/parse.py` with a Typer subcommand group for config parsing.

Commands:
1. `junos <config-file-or-stdin>` — Parse JunOS JSON config
   - Accept config from file path or stdin
   - `--network-os` option (default "juniper_junos")
   - Output full ParsedConfig as JSON or summarized table
2. `list-parsers` — List available parsers

```python
import typer
import json
import sys
from nautobot_mcp.parsers import ParserRegistry

app = typer.Typer(name="parse", help="Parse device configurations into structured data.")

@app.command()
def junos(
    config_file: str = typer.Argument(None, help="Path to JunOS JSON config file. Use '-' for stdin."),
    network_os: str = typer.Option("juniper_junos", help="Parser network_os identifier"),
    json_output: bool = typer.Option(False, "--json", help="Output full JSON"),
):
    """Parse a JunOS configuration (JSON format) into structured data."""
    if config_file == "-" or config_file is None:
        config_text = sys.stdin.read()
    else:
        with open(config_file) as f:
            config_text = f.read()
    
    config_data = json.loads(config_text)
    parser = ParserRegistry.get(network_os)
    result = parser.parse(config_data)
    
    if json_output:
        print(result.model_dump_json(indent=2))
    else:
        # Summary table
        print(f"Hostname: {result.hostname}")
        print(f"Platform: {result.platform} ({result.network_os})")
        print(f"Interfaces: {len(result.interfaces)}")
        print(f"IP Addresses: {len(result.ip_addresses)}")
        print(f"VLANs: {len(result.vlans)}")
        print(f"Routing Instances: {len(result.routing_instances)}")
        print(f"Protocols: {len(result.protocols)}")
        print(f"Firewall Filters: {len(result.firewall_filters)}")
        if result.warnings:
            print(f"Warnings: {len(result.warnings)}")
            for w in result.warnings:
                print(f"  ⚠ {w}")

@app.command()
def list_parsers():
    """List available configuration parsers."""
    parsers = ParserRegistry.list_parsers()
    for p in parsers:
        print(f"  - {p}")
```
